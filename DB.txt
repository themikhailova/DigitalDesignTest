2. 
WITH RECURSIVE ChiefSequence AS (
    SELECT EMPLOYEE.ID, EMPLOYEE.CHIEF_ID, 1 AS Level
    FROM EMPLOYEE
    WHERE EMPLOYEE.CHIEF_ID IS NULL -- начальник
    
    UNION ALL
    
    SELECT EMPLOYEE.ID, EMPLOYEE.CHIEF_ID, ChiefSequence.Level + 1
    FROM EMPLOYEE
    INNER JOIN ChiefSequence ON EMPLOYEE.CHIEF_ID = ChiefSequence.ID
)
SELECT MAX(Level) AS MaxLevel
FROM ChiefSequence;

3.
SELECT
    DEPARTMENT.NAME AS Department,
    MAX(TotalSalary) AS MaxTotalSalary
FROM (
    SELECT
        DEPARTMENT.NAME,
        SUM(EMPLOYEE.SALARY) AS TotalSalary
    FROM
        EMPLOYEE
    JOIN
        DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.ID
    GROUP BY
        DEPARTMENT.NAME
) AS DepartmentSalary
